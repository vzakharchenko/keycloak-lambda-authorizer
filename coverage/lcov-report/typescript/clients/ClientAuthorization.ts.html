  clientJWT(payload: any, privateKey: RSAKey): Promise<string> {
    return new Promise((resolve, reject) => {
      jws.createSign({
        header: {alg: 'RS256', typ: 'RSA'},
        privateKey,
        payload,
      }).on('done', (signature) => {
        resolve(signature);
      }).on('error', (e) => {
        this.options.logger.log(`error:${e}`);
        reject(e);
      });
    });
  }
